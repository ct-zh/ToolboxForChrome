function setupEventListeners() { const timestampElement = document.getElementById('currentTimestamp'); if (timestampElement) { // 点击复制 timestampElement.addEventListener('click', () => { const timestamp = timestampElement.textContent; navigator.clipboard.writeText(timestamp).then(() => { alert('时间戳已复制: ' + timestamp); }); }); // 每秒更新 setInterval(updateTimestamp, 1000); updateTimestamp(); } } // Wait for DOM to load document.addEventListener('DOMContentLoaded', setupEventListeners);
